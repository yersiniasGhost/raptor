const newest = socHistory[unitId][socHistory[unitId].length - 1]; // Calculate SOC change per hour const elapsedHours = (newest.timestamp - oldReading.timestamp) / (1000 * 60 * 60); const socChange = newest.soc - oldReading.soc;
